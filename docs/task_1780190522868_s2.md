### 2.2 公開関数の実装仕様

#### `collectGenreSeeds(genre, options)`

```
目的   : 指定ジャンルの hit/miss シード動画を収集してJSONに保存する
引数   : genre     - 'VTuber' | 'clip' | 'music' | 'realshot' | 'gaming'
         options   - CollectOptions（Phase 1 §3.3 参照）
戻り値 : CollectResult（Phase 1 §3.3 参照）

処理フロー:
  1. validateApiKey()   ← YOUTUBE_API_KEY 存在チェック
  2. estimateQuotaCost() で消費量を算出
  3. getRemainingQuota() で当日残量を確認
     → 不足なら { error: 'QUOTA_INSUFFICIENT' } を含む CollectResult を返す
  4. _collectHit(genre, options.hitCount)  ← hit動画収集
  5. _collectMiss(genre, options.missCount, hitChannelIds)  ← miss動画収集
  6. CollectResult を組み立てて return

エラー時:
  - quotaExceeded: 即座に停止し保存済みデータを維持
  - その他: errors[] に push して処理継続（動画単位のスキップ）
```

#### `collectGenreSeedsWithProgress(genre, onProgress)`

```
目的   : 進捗コールバック付き収集（Discord の !collect-seeds コマンドから呼ぶ）
引数   : genre      - ジャンル文字列
         onProgress - (message: string) => void
         ※ options は内部でデフォルト値を使用（hitCount: 100, missCount: 100）

進捗通知タイミング:
  - 収集開始時: "📡 [VTuber] hit動画収集開始..."
  - Search 各ページ取得後: "🔍 候補 {n}件取得中..."
  - Videos.list バッチ後: "📊 統計取得中 ({n}/{total})..."
  - hit フィルタ後: "✅ hit: {n}件確定"
  - miss 収集開始時: "🔍 [VTuber] miss動画収集開始..."
  - miss フィルタ後: "✅ miss: {n}件確定"
  - 保存完了: "💾 保存完了 hit:{h}件 miss:{m}件 / クォータ消費: {u} units"
```

#### `loadSeeds(genre)`

```
目的   : 保存済みシードデータを読み込む
引数   : genre - ジャンル文字列
戻り値 : { hit: SeedVideo[], miss: SeedVideo[] }
実装   : SEEDS_DIR/{genre}/hit/*.json と miss/*.json を fs.readdirSync で読み込み
         JSON.parse して配列に push
         ファイルが存在しない場合は { hit: [], miss: [] } を返す
```

#### `estimateQuotaCost(genre, options)`

```
目的   : 実行前クォータ消費見積もり
引数   : genre, options
戻り値 : { estimatedUnits: number, details: { search, videos, channels } }

計算式:
  hitCandidates  = options.hitCount * 3        // フィルタ脱落を考慮して3倍
  missCandidates = options.missCount * 2
  searchCalls    = Math.ceil(hitCandidates / MAX_RESULTS) * hitKeywords.length
                 + Math.ceil(missCandidates / MAX_RESULTS) * missKeywords.length
  videoCalls     = Math.ceil((hitCandidates + missCandidates) / MAX_RESULTS)
  channelCalls   = Math.ceil(hitCandidates / MAX_RESULTS)   // miss は hit時取得済み

  estimatedUnits = searchCalls * 100 + videoCalls * 1 + channelCalls * 1
```

#### `getSeedStats()`

```
目的   : 全ジャンルの収集済み件数を返す
戻り値 : { VTuber: { hit: 87, miss: 94 }, clip: { hit: 0, miss: 0 }, ... }
実装   : SEEDS_DIR を fs.readdirSync でスキャン
         各ジャンルの hit/ miss/ ディレクトリのファイル数をカウント
```

### 2.4 データ変換関数

#### `parseDuration(iso8601)`

```javascript
// "PT12M34S" → 754
// "PT1H2M3S" → 3723
// "PT45S"    → 45
function parseDuration(iso8601) {
  const m = iso8601.match(/PT(?:(\d+)H)?(?:(\d+)M)?(?:(\d+)S)?/);
  if (!m) return 0;
  return (parseInt(m[1] || 0) * 3600)
       + (parseInt(m[2] || 0) * 60)
       + parseInt(m[3] || 0);
}
```

#### `calcBuzzRatio(viewCount, subscriberCount)`

```javascript
// 登録者数下限未満はスキップ → null を返す
function calcBuzzRatio(viewCount, subscriberCount) {
  if (subscriberCount < MIN_SUBSCRIBERS) return null;
  return viewCount / subscriberCount;
}
```

```
戻り値: SeedVideo（Phase 1 §3.3 参照）
```

### 2.5 フィルタ関数

```javascript
// buzzRatio >= HIT_THRESHOLD のもの、上位 count 件
function filterHit(videos, count) {
  return videos
    .filter(v => v.buzzRatio !== null && v.buzzRatio >= HIT_THRESHOLD)
    .sort((a, b) => b.buzzRatio - a.buzzRatio)
    .slice(0, count);
}

// buzzRatio <= MISS_THRESHOLD のもの、上位 count 件
// hiddenSubscriberCount は calcBuzzRatio 段階で null 化済みのため不要
function filterMiss(videos, count) {
  return videos
    .filter(v => v.buzzRatio !== null && v.buzzRatio <= MISS_THRESHOLD)
    .sort((a, b) => a.buzzRatio - b.buzzRatio)
    .slice(0, count);
}
```

```javascript
// 保存: 動画1件 = 1ファイル（videoId.json）
async function saveSeeds(genre, label, videos) {
  const dir = path.join(SEEDS_DIR, genre, label);
  fs.mkdirSync(dir, { recursive: true });
  for (const v of videos) {
    const p = path.join(dir, `${v.videoId}.json`);
    fs.writeFileSync(p, JSON.stringify(v, null, 2), 'utf-8');
  }
}

// 読み込み: ディレクトリ内の全JSONを配列で返す
function loadSeedsRaw(genre, label) {
  const dir = path.join(SEEDS_DIR, genre, label);
  if (!fs.existsSync(dir)) return [];
  return fs.readdirSync(dir)
    .filter(f => f.endsWith('.json'))
    .map(f => JSON.parse(fs.readFileSync(path.join(dir, f), 'utf-8')));
}

// 既存ファイルの videoId セットを返す（重複スキップ用）
function getExistingVideoIds(genre, label) {
  const dir = path.join(SEEDS_DIR, genre, label);
  if (!fs.existsSync(dir)) return new Set();
  return new Set(
    fs.readdirSync(dir)
      .filter(f => f.endsWith('.json'))
      .map(f => f.replace('.json', ''))
  );
}
```

### 2.7 クォータ管理関数

```javascript
// 当日消費 units を _quota_log.json に加算
function incrementQuotaCounter(units) {
  const today = new Date().toISOString().slice(0, 10);
  let log = { date: today, totalUnitsUsed: 0, lastUpdated: '' };
  if (fs.existsSync(QUOTA_LOG_PATH)) {
    const saved = JSON.parse(fs.readFileSync(QUOTA_LOG_PATH, 'utf-8'));
    if (saved.date === today) log = saved;
    // 日付が変わっていれば log はリセット（初期値のまま使用）
  }
  log.totalUnitsUsed += units;
  log.lastUpdated = new Date().toISOString();
  fs.mkdirSync(path.dirname(QUOTA_LOG_PATH), { recursive: true });
  fs.writeFileSync(QUOTA_LOG_PATH, JSON.stringify(log, null, 2), 'utf-8');
}

// 本日の残クォータ推定（10,000 units 固定上限）
function getRemainingQuota() {
  const today = new Date().toISOString().slice(0, 10);
  if (!fs.existsSync(QUOTA_LOG_PATH)) return 10000;
  const log = JSON.parse(fs.readFileSync(QUOTA_LOG_PATH, 'utf-8'));
  if (log.date !== today) return 10000;
  return 10000 - log.totalUnitsUsed;
}
```

### 2.8 リトライ・バリデーション

```javascript
async function withRetry(fn, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (err) {
      // quotaExceeded はリトライしない
      if (err.code === 403 && err.errors?.[0]?.reason === 'quotaExceeded') throw err;
      if (i === maxRetries - 1) throw err;
      await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000));
    }
  }
}
```

```javascript
// ─── !collect-seeds ─────────────────────────────────────
case '!collect-seeds': {
  if (!security.isOwner(message.author.id, DISCORD_OWNER_ID)) {
    await message.reply('❌ このコマンドはオーナーのみ実行できます。');
    break;
  }
  if (!ENABLE_SEED_COLLECTOR) {
    await message.reply('❌ シード収集は無効です（`ENABLE_SEED_COLLECTOR=true` を設定してください）。');
    break;
  }

  const subCmd = args[0] || '';

  // !collect-seeds status
  if (subCmd === 'status') {
    const stats = seedCollector.getSeedStats();
    const lines = Object.entries(stats)
      .map(([g, s]) => `**${g}**: hit ${s.hit}件 / miss ${s.miss}件`);
    await message.reply(`**シード収集状況:**\n${lines.join('\n')}`);
    break;
  }

  // !collect-seeds estimate [genre]
  if (subCmd === 'estimate') {
    const genre = args[1];
    const genres = genre ? [genre] : ['VTuber', 'clip', 'music', 'realshot', 'gaming'];
    const lines = genres.map(g => {
      const { estimatedUnits } = seedCollector.estimateQuotaCost(g, {});
      return `**${g}**: 約 ${estimatedUnits} units`;
    });
    await message.reply(`**クォータ消費見積もり:**\n${lines.join('\n')}`);
    break;
  }

  // !collect-seeds all または !collect-seeds <genre>
  const VALID_GENRES = ['VTuber', 'clip', 'music', 'realshot', 'gaming'];
  const targetGenres = (subCmd === 'all')
    ? VALID_GENRES
    : VALID_GENRES.includes(subCmd) ? [subCmd] : null;

  if (!targetGenres) {
    await message.reply(
      '❌ 使い方:\n' +
      '`!collect-seeds <genre>` — 指定ジャンルを収集\n' +
      '`!collect-seeds all` — 全ジャンルを収集\n' +
      '`!collect-seeds status` — 収集済み件数\n' +
      '`!collect-seeds estimate` — クォータ見積もり\n\n' +
      `有効なジャンル: ${VALID_GENRES.join(' / ')}`
    );
    break;
  }

  await message.reply(`🚀 シード収集を開始します: **${targetGenres.join(', ')}**`);

  for (const genre of targetGenres) {
    try {
      await seedCollector.collectGenreSeedsWithProgress(genre, async (msg) => {
        await message.channel.send(msg);
      });
    } catch (err) {
      await message.channel.send(`❌ [${genre}] 収集エラー: ${err.message}`);
      if (err.message.includes('quotaExceeded')) break;  // クォータ切れは全停止
    }
  }
  break;
}
```
